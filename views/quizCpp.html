Q 5 - What is the output of the following program?

            #include<iostream>
            
            using namespace std;
            class Base {
            public:
               void f() {
                  cout<<"Base\n";
               }
            };
            class Derived:public Base {
            public:
               void f() {
                  cout<<"Derived\n";
               }
            };
            main() { 
               Derived obj; 
               obj.Base::f();
            }

A - Base

B - Derived

C - Compile error

D - None of the above.

Answer : A

Explaination

The method f() inherited from Base is referred using :: operator.

            #include<iostream>
            
            using namespace std;
            class Base {
            public:
               void f() {
                  cout<<"Base\n";
               }
            };
            class Derived:public Base {
            public:
               void f() {
                  cout<<"Derived\n";
               }
            };
            main() { 
               Derived obj; 
               obj.Base::f();
            }

Q 8 - What is the output of the following program?

            #include<iostream>
            
            using namespace std;
            void swap(int m, int n) {
               int x = m;
            
                m = n;
                n = x;
            }
            main() {
               int x = 5, y = 3;
               
                swap(x,y);
                cout<<x<<" "<<y;
            }

A - 3 5

B - 5 3

C - 5 5

D - Compile error

Answer : B

Explaination

5 3, call by value mechanism can’t alter actual arguments.

            #include<iostream>
            
            using namespace std;
            void swap(int m, int n) {
               int x = m;
            
                m = n;
                n = x;
            }

               int x = 5, y = 3;

                cout<<x<<" "<<y;
            }

Q 9 - An inline function can execute faster than a normal function.

B - False
